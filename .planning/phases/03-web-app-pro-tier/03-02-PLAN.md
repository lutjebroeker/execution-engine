    **Step 2 — Create src/lib/stripe.ts:**
    ```typescript
    import { loadStripe } from '@stripe/stripe-js';

    let stripePromise: ReturnType<typeof loadStripe>;

    export function getStripe() {
      if (!stripePromise) {
        stripePromise = loadStripe(import.meta.env.VITE_STRIPE_PUBLISHABLE_KEY);
      }
      return stripePromise;
    }
    ```
    VITE_STRIPE_PUBLISHABLE_KEY is the only Stripe key in the client bundle. Never STRIPE_SECRET_KEY.

    **Step 3 — Create supabase/functions/create-checkout-session/index.ts:**
    Deno Edge Function. Pattern from research:
    - Import: `import Stripe from 'npm:stripe'`
    - Import: `import { createClient } from 'npm:@supabase/supabase-js'`
    - Handle OPTIONS for CORS: return `new Response('ok', { headers: corsHeaders })`
    - corsHeaders: `{ 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type' }`
    - Get user JWT from Authorization header, verify with `supabase.auth.getUser()` (NOT getSession)
    - If no user: return 401
    - Parse body: `{ priceId, returnUrl }`
    - priceId defaults to `Deno.env.get('STRIPE_PRICE_ID')` if not provided
    - Create Stripe Checkout session: `mode: 'subscription'`, `line_items: [{ price: priceId, quantity: 1 }]`, `success_url: returnUrl + '?success=true'`, `cancel_url: returnUrl + '?canceled=true'`, `customer_email: user.email`, `metadata: { user_id: user.id }`
    - Return: `{ url: session.url }`

    **Step 4 — Create supabase/functions/stripe-webhook/index.ts:**
    Deno Edge Function. No CORS needed (called by Stripe, not browser).
    - Raw body text: `const body = await req.text()`
    - Signature: `req.headers.get('stripe-signature')`
    - Verify: `stripe.webhooks.constructEventAsync(body, signature, webhookSecret)`
    - On bad signature: return 400
    - Admin Supabase client using `SUPABASE_SERVICE_ROLE_KEY` (bypasses RLS — required for webhook which has no user JWT)
    - Handle `checkout.session.completed`: update profiles SET `subscription_status = 'active'`, `stripe_customer_id = session.customer` WHERE id = `session.metadata.user_id`
    - Handle `customer.subscription.updated`: update `subscription_status` based on `subscription.status` ('active', 'past_due', 'canceled')
    - Handle `customer.subscription.deleted`: update `subscription_status = 'canceled'` WHERE `stripe_customer_id = subscription.customer`
    - Return 200 `{ received: true }` for all handled events

    **Step 5 — Create supabase/functions/create-portal-session/index.ts:**
    Deno Edge Function with CORS.
    - Verify user JWT via `supabase.auth.getUser()`
    - Fetch `stripe_customer_id` from profiles using admin client (service role)
    - If no `stripe_customer_id`: return 400 "No Stripe customer found"
    - Create billing portal session: `stripe.billingPortal.sessions.create({ customer: stripeCustomerId, return_url: returnUrl })`
    - Return: `{ url: session.url }`

    **Step 6 — Create src/components/ProGate.tsx:**
    ```typescript
    interface ProGateProps {
      children: React.ReactNode;
      feature?: string; // Optional label: "cloud sync", "AI coaching"
    }

    export function ProGate({ children, feature }: ProGateProps) {
      const { isPro, isLoading } = useSubscription();
      if (isLoading) return null; // Don't flash upgrade prompt while loading
      if (isPro) return <>{children}</>;
      return (
        <div className="ee-card text-center py-8">
          <p className="font-georgia text-lg font-bold text-[#1a1a2e] mb-2">Pro feature</p>
          <p className="text-[#6B7280] text-sm mb-4">
            {feature ? `${feature} is` : 'This is'} available on the Pro plan (€19.99/month).
          </p>
          <button onClick={handleUpgrade} className="bg-[#1D4ED8] text-white px-6 py-2 rounded-lg text-sm font-medium hover:bg-[#1e40af]">
            Upgrade to Pro
          </button>
        </div>
      );
    }
    ```
    `handleUpgrade` calls `supabase.functions.invoke('create-checkout-session', { body: { returnUrl: window.location.origin + '/instellingen' } })` then sets `window.location.href = data.url`.
