    ```typescript
    async function handleModeToggle(enabled: boolean) {
      if (!enabled) {
        // Switching back to Cloud
        setObsidianMode(false);
        toast({
          title: 'Switched to Cloud sync',
          description: "Edits made directly in Obsidian won't sync back to the app.",
        });
        return;
      }
      // Switching to Obsidian mode — just enable, folder picker shown in UI
      setObsidianMode(true);
      toast({ title: 'Obsidian mode enabled', description: 'Select your vault folder to start syncing.' });
    }
    ```

    **handleConnect function:**
    ```typescript
    async function handleConnect() {
      try {
        const vaultName = await connect();
        toast({ title: `Connected: ${vaultName}`, description: 'Syncing existing data to vault...' });
        // Backfill is fire-and-forget — do not await in the click handler critical path
        backfill()
          .then(({ written, errors }) => {
            toast({
              title: 'Vault sync complete',
              description: `${written} notes written${errors > 0 ? `, ${errors} errors (check console)` : ''}.`,
            });
          })
          .catch(err => console.warn('[vault backfill error]', err));
      } catch (err) {
        // User cancelled picker or API unavailable
        console.warn('[vault connect]', err);
      }
    }
    ```
