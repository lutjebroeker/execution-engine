    ```typescript
    import type { FileSystemDirectoryHandle } from 'native-file-system-adapter';  // types only; use browser native

    const IDB_KEY = 'vault-directory-handle';

    export async function saveHandle(handle: FileSystemDirectoryHandle): Promise<void> {
      await set(IDB_KEY, handle);
    }

    export async function loadHandle(): Promise<FileSystemDirectoryHandle | undefined> {
      return get(IDB_KEY);
    }

    export async function clearHandle(): Promise<void> {
      await del(IDB_KEY);
    }

    export function isVaultApiSupported(): boolean {
      return typeof window !== 'undefined' && 'showDirectoryPicker' in window;
    }
    ```

    **writeNote(relativePath, content):**
    ```typescript
    const writeNote = async (relativePath: string, content: string): Promise<void> => {
      if (!dirHandle) return;
      // Always verify permission before write
      const perm = await dirHandle.queryPermission({ mode: 'readwrite' });
      const effectivePerm = perm === 'granted'
        ? perm
        : await dirHandle.requestPermission({ mode: 'readwrite' });
      if (effectivePerm !== 'granted') return; // silent fail — Supabase has the data

      const parts = relativePath.split('/');
      const filename = parts.pop()!;
      let current: FileSystemDirectoryHandle = dirHandle;
      for (const part of parts) {
        current = await current.getDirectoryHandle(part, { create: true });
      }
      const fileHandle = await current.getFileHandle(filename, { create: true });
      const writable = await fileHandle.createWritable();
      try {
        await writable.write(content);
      } finally {
        await writable.close();
      }
    };
    ```

    Anti-patterns (from RESEARCH.md — do NOT violate):
    - Never call requestPermission() in useEffect on mount (requires user gesture)
    - Never call showDirectoryPicker() outside a button onClick
    - Never await vault writes in the Supabase mutation critical path

    **useVaultBackfill hook:**
    ```typescript
    export function useVaultBackfill() {
      const { writeNote } = useVaultContext();
      const { user } = useAuth();

      const backfill = async (): Promise<{ written: number; errors: number }> => {
        if (!user) return { written: 0, errors: 0 };
        const uid = user.id;

        // Fetch all data types in parallel
        const [checkInsRes, reflectionsRes, sessionsRes, reviewsRes, visionRes, cyclesRes, plansRes] = await Promise.all([
          supabase.from('daily_notes').select('*').eq('user_id', uid),
          supabase.from('weekly_reflections').select('*').eq('user_id', uid),
          supabase.from('flow_sessions').select('*').eq('user_id', uid),
          supabase.from('twelve_week_reviews').select('*').eq('user_id', uid),
          supabase.from('vision').select('*').eq('user_id', uid).maybeSingle(),
          supabase.from('cycles').select('*').eq('user_id', uid),
          supabase.from('weekly_plans').select('*').eq('user_id', uid),
        ]);

        // Build a cycle map for start_date lookups
        const cycleMap = new Map((cyclesRes.data ?? []).map(c => [c.id, c]));

        // Collect all writes — include morning and evening from daily_notes
        const writes: Array<{ path: string; content: string }> = [];

        for (const row of (checkInsRes.data ?? [])) {
          if (row.morning_checkin) {
            const note = buildCheckInNote({ id: row.id + '-morning', date: row.date, type: 'morning', data: row.morning_checkin });
            writes.push(note);
          }
          if (row.evening_reflection) {
            const note = buildCheckInNote({ id: row.id + '-evening', date: row.date, type: 'evening', data: row.evening_reflection });
            writes.push(note);
          }
        }

        for (const row of (sessionsRes.data ?? [])) {
          writes.push(buildFlowSessionNote({
            id: row.id,
            startTime: row.start_time,
            duration: row.duration_minutes ?? 0,
            trigger: row.trigger,
            taskType: row.task_type,
            notes: row.notes,
          }));
        }

        for (const row of (reflectionsRes.data ?? [])) {
          const cycle = cycleMap.get(row.cycle_id);
          if (!cycle) continue;
          writes.push(buildReflectionNote({
            weekNumber: row.week_number,
            cycleId: row.cycle_id,
            completionScore: row.completion_score,
            whatWorkedWell: row.what_worked_well,
            whatDidntWork: row.what_didnt_work,
            whatDistracted: row.what_distracted,
            whatMustChange: row.what_must_change,
          }, cycle.start_date));
        }

        for (const row of (plansRes.data ?? [])) {
          const cycle = cycleMap.get(row.cycle_id);
          if (!cycle) continue;
          writes.push(buildWeeklyPlanNote({ weekNumber: row.week_number, plan: row.plan }, cycle.start_date));
        }

        for (const cycle of (cyclesRes.data ?? [])) {
          for (const goal of (cycle.goals ?? [])) {
            writes.push(buildGoalNote(goal, cycle.start_date, cycle.end_date));
          }
        }

        for (const row of (reviewsRes.data ?? [])) {
          const cycle = cycleMap.get(row.cycle_id);
          if (!cycle) continue;
          writes.push(buildReviewNote({
            cycleId: row.cycle_id,
            biggestWins: row.biggest_wins,
            biggestChallenges: row.biggest_challenges,
            keyLessons: row.key_lessons,
            nextCyclePriorities: row.next_cycle_priorities,
          }, cycle.start_date));
        }

        if (visionRes.data) {
          writes.push(buildVisionNote(visionRes.data));
        }

        // Sequential writes to avoid overwhelming FS API
        let written = 0;
        let errors = 0;
        for (const { path, content } of writes) {
          try {
            await writeNote(path, content);
            written++;
          } catch (e) {
            console.warn('[vault backfill]', path, e);
            errors++;
          }
        }

        return { written, errors };
      };

      return { backfill };
    }
    ```
